    <script>
        // Initialize Lucide Icons
        lucide.createIcons();

        // Haptic Feedback for Primary Buttons
        document.querySelectorAll('.btn-action.primary').forEach(btn => {
            btn.addEventListener('click', () => {
                if (window.navigator.vibrate) {
                    window.navigator.vibrate(12);
                }
            });
        });

        // Tab Switching Logic
        function switchTab(element) {
            document.querySelectorAll('.tab-item').forEach(tab => {
                tab.classList.remove('active');
            });
            element.classList.add('active');
            
            // Subtle feedback
            if (window.navigator.vibrate) { window.navigator.vibrate(5); }
        }

        // Animation on entry
        document.addEventListener('DOMContentLoaded', () => {
            const elements = document.querySelectorAll('.crystal-surface, header, nav');
            elements.forEach((el, i) => {
                el.style.opacity = '0';
                el.style.transform = 'translateY(15px)';
                setTimeout(() => {
                    el.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    el.style.opacity = '1';
                    el.style.transform = 'translateY(0)';
                }, i * 80);
            });
        });
    </script>
