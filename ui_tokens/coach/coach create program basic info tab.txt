    <script>
        // Initialize Lucide icons
        lucide.createIcons();

        // Subtle animation for elements on load
        document.addEventListener('DOMContentLoaded', () => {
            const elements = document.querySelectorAll('.crystal-card, header, .tab-btn');
            elements.forEach((el, index) => {
                el.style.opacity = '0';
                el.style.transform = 'translateY(15px)';
                setTimeout(() => {
                    el.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    el.style.opacity = '1';
                    el.style.transform = 'translateY(0)';
                }, index * 80);
            });
        });

        // Haptic feedback simulation
        document.querySelectorAll('button').forEach(btn => {
            btn.addEventListener('click', () => {
                if (window.navigator.vibrate) {
                    window.navigator.vibrate(8);
                }
            });
        });

        // Form Submit Prevent Default
        document.getElementById('programForm').addEventListener('submit', (e) => {
            e.preventDefault();
            const btn = e.target.querySelector('.btn-primary');
            btn.innerHTML = '<i data-lucide="loader-2" class="w-5 h-5 animate-spin"></i> Saving...';
            lucide.createIcons();
            
            setTimeout(() => {
                btn.innerHTML = 'Saved Successfully <i data-lucide="check" class="w-5 h-5"></i>';
                btn.classList.remove('btn-primary');
                btn.classList.add('bg-accent', 'text-white');
                lucide.createIcons();
            }, 1200);
        });
    </script>
