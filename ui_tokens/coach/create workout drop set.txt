    <script>
        // Initialize Lucide icons
        lucide.createIcons();

        // Search Dropdown Toggle
        const searchInput = document.getElementById('exercise-search');
        const results = document.getElementById('results');

        searchInput.addEventListener('focus', () => {
            results.style.display = 'block';
        });

        document.addEventListener('click', (e) => {
            if (!searchInput.contains(e.target) && !results.contains(e.target)) {
                results.style.display = 'none';
            }
        });

        const items = document.querySelectorAll('.search-item');
        items.forEach(item => {
            item.addEventListener('click', () => {
                searchInput.value = item.innerText;
                results.style.display = 'none';
            });
        });

        // Save Feedback
        const saveBtn = document.getElementById('save-btn');
        saveBtn.addEventListener('click', () => {
            const originalText = saveBtn.innerHTML;
            saveBtn.innerHTML = '<i data-lucide="check" class="w-5 h-5"></i> Added!';
            saveBtn.style.background = 'linear-gradient(135deg, #10B981 0%, #059669 100%)';
            saveBtn.style.boxShadow = '0 4px 16px rgba(16, 185, 129, 0.3)';
            lucide.createIcons();

            if (window.navigator.vibrate) {
                window.navigator.vibrate([20, 10, 20]);
            }

            setTimeout(() => {
                saveBtn.innerHTML = originalText;
                saveBtn.style.background = '';
                saveBtn.style.boxShadow = '';
                lucide.createIcons();
            }, 2000);
        });

        // Entrance Animation
        document.addEventListener('DOMContentLoaded', () => {
            const card = document.querySelector('.crystal-card');
            card.style.opacity = '0';
            card.style.transform = 'translateY(30px)';
            setTimeout(() => {
                card.style.transition = 'all 0.8s cubic-bezier(0.2, 0.8, 0.2, 1)';
                card.style.opacity = '1';
                card.style.transform = 'translateY(0)';
            }, 100);
        });
    </script>
