        lucide.createIcons();

        // Subtle animation for cards on load
        document.addEventListener('DOMContentLoaded', () => {
            const elements = document.querySelectorAll('.crystal-card, .calendar-day');
            elements.forEach((el, index) => {
                el.style.opacity = '0';
                el.style.transform = 'translateY(10px)';
                setTimeout(() => {
                    el.style.transition = 'all 0.5s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    el.style.opacity = '1';
                    el.style.transform = 'translateY(0)';
                }, index * 20);
            });
        });

        // Haptic feedback simulation
        document.querySelectorAll('button, .calendar-day').forEach(btn => {
            btn.addEventListener('click', () => {
                if (window.navigator.vibrate) window.navigator.vibrate(8);
            });
        });

        // Day Selection Logic
        document.querySelectorAll('.calendar-day:not(.dimmed)').forEach(day => {
            day.addEventListener('click', function() {
                document.querySelectorAll('.calendar-day').forEach(d => d.classList.remove('active'));
                this.classList.add('active');
            });
        });
