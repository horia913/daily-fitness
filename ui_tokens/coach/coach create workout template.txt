    <script>
        // Initialize Lucide
        lucide.createIcons();

        let activeProtocol = 'Straight Set';

        function selectProtocol(el) {
            document.querySelectorAll('.block-tile').forEach(tile => tile.classList.remove('active'));
            el.classList.add('active');
            activeProtocol = el.querySelector('span').innerText;
            
            // Haptic feedback simulation
            if (window.navigator.vibrate) window.navigator.vibrate(10);
        }

        function addExercise(name, muscle) {
            const canvas = document.getElementById('builder-canvas');
            const emptyState = document.getElementById('empty-canvas');
            const template = document.getElementById('exercise-card-template');
            
            if (emptyState) emptyState.style.display = 'none';

            const clone = template.content.cloneNode(true);
            clone.querySelector('.exercise-title').innerText = name;
            clone.querySelector('.block-label').innerText = activeProtocol;

            // Simple visual cue for supersets
            if (activeProtocol === 'Superset') {
                clone.querySelector('.crystal-card').style.borderLeftColor = '#F97316';
                clone.querySelector('.block-label').classList.remove('bg-secondary/20', 'text-secondary');
                clone.querySelector('.block-label').classList.add('bg-orange-500/20', 'text-orange-400');
            }

            canvas.appendChild(clone);
            lucide.createIcons();

            // Vibrate on add
            if (window.navigator.vibrate) window.navigator.vibrate(20);
            
            // Scroll to bottom of canvas
            window.scrollTo({ top: document.body.scrollHeight, behavior: 'smooth' });
        }

        // Search Filter Logic
        document.querySelector('aside input').addEventListener('input', (e) => {
            const term = e.target.value.toLowerCase();
            const items = document.querySelectorAll('.exercise-item');
            
            items.forEach(item => {
                const text = item.innerText.toLowerCase();
                item.style.display = text.includes(term) ? 'flex' : 'none';
            });
        });

        // Animation Entrance
        document.addEventListener('DOMContentLoaded', () => {
            const cards = document.querySelectorAll('.crystal-card');
            cards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 50);
            });
        });
    </script>
