    <script>
        // Initialize Lucide Icons
        lucide.createIcons();

        // Interaction Feedback
        document.querySelectorAll('button').forEach(btn => {
            btn.addEventListener('click', () => {
                if (window.navigator.vibrate) window.navigator.vibrate(10);
            });
        });

        // Entry Reveal Animation
        document.addEventListener('DOMContentLoaded', () => {
            const sections = document.querySelectorAll('section');
            sections.forEach((s, i) => {
                s.style.opacity = '0';
                s.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    s.style.transition = 'all 0.8s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    s.style.opacity = '1';
                    s.style.transform = 'translateY(0)';
                }, i * 150);
            });
        });

        // Simple Range Sync Simulation
        const sliders = document.querySelectorAll('.macro-slider');
        sliders.forEach(slider => {
            slider.addEventListener('input', (e) => {
                const val = e.target.value;
                const display = e.target.previousElementSibling.querySelector('.mono');
                if (display) display.textContent = val + '%';
            });
        });
