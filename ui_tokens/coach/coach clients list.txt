    <script>
        // Initialize Lucide icons
        lucide.createIcons();

        // Selection Logic
        const checkboxes = document.querySelectorAll('.client-select');
        const bulkBar = document.getElementById('bulk-bar');
        const selectCount = document.getElementById('select-count');

        checkboxes.forEach(box => {
            box.addEventListener('change', () => {
                const checked = document.querySelectorAll('.client-select:checked');
                const count = checked.length;
                
                if (count > 0) {
                    bulkBar.classList.add('visible');
                    selectCount.innerText = count;
                } else {
                    bulkBar.classList.remove('visible');
                }
            });
        });

        function clearSelection() {
            checkboxes.forEach(box => box.checked = false);
            bulkBar.classList.remove('visible');
        }

        // Animation for entry
        document.addEventListener('DOMContentLoaded', () => {
            const cards = document.querySelectorAll('.crystal-card');
            cards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 100);
            });
        });

        // Haptic feedback simulation
        document.querySelectorAll('button').forEach(btn => {
            btn.addEventListener('click', () => {
                if (window.navigator.vibrate) window.navigator.vibrate(10);
            });
        });
    </script>
