    <script>
        // Initialize Lucide icons
        lucide.createIcons();

        // UI Logic
        const checkboxes = document.querySelectorAll('.client-checkbox');
        const countSpan = document.getElementById('selection-count');
        const btnCount = document.getElementById('btn-count');
        const selectAllBtn = document.getElementById('select-all');
        const assignBtn = document.getElementById('assign-btn');

        function updateCounts() {
            const checkedCount = document.querySelectorAll('.client-checkbox:checked').length;
            countSpan.textContent = `${checkedCount} CLIENTS SELECTED`;
            btnCount.textContent = checkedCount;

            // Haptic Feedback for Mobile
            if (window.navigator.vibrate) {
                window.navigator.vibrate(5);
            }
        }

        checkboxes.forEach(cb => {
            cb.addEventListener('change', updateCounts);
        });

        selectAllBtn.addEventListener('click', () => {
            const allChecked = Array.from(checkboxes).every(cb => cb.checked);
            checkboxes.forEach(cb => cb.checked = !allChecked);
            updateCounts();
            selectAllBtn.textContent = !allChecked ? 'Deselect All' : 'Select All';
        });

        // Search filtering simulation
        const searchInput = document.querySelector('input[type="text"]');
        const clientItems = document.querySelectorAll('.client-item');

        searchInput.addEventListener('input', (e) => {
            const term = e.target.value.toLowerCase();
            clientItems.forEach(item => {
                const name = item.querySelector('.font-bold').textContent.toLowerCase();
                const visible = name.includes(term);
                item.style.display = visible ? 'flex' : 'none';
            });
        });

        // Assignment Celebration
        assignBtn.addEventListener('click', () => {
            const count = document.querySelectorAll('.client-checkbox:checked').length;
            if(count === 0) {
                alert('Please select at least one client.');
                return;
            }
            
            assignBtn.style.opacity = '0.7';
            assignBtn.textContent = "ASSIGNING...";
            
            setTimeout(() => {
                assignBtn.innerHTML = '<i data-lucide="check-circle" class="w-6 h-6"></i> SUCCESSFULLY ASSIGNED';
                assignBtn.classList.remove('btn-assign');
                assignBtn.style.background = '#10B981';
                assignBtn.style.opacity = '1';
                lucide.createIcons();
            }, 1000);
        });

        // Animation on load
        document.addEventListener('DOMContentLoaded', () => {
            const sections = document.querySelectorAll('section, header');
            sections.forEach((s, i) => {
                s.style.opacity = '0';
                s.style.transform = 'translateY(15px)';
                setTimeout(() => {
                    s.style.transition = 'all 0.5s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    s.style.opacity = '1';
                    s.style.transform = 'translateY(0)';
                }, i * 80);
            });
        });
    </script>
