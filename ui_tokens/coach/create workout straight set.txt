    <script>
        // Initialize Lucide
        lucide.createIcons();

        // Haptic Feedback simulation
        document.querySelectorAll('button, input[type="range"]').forEach(el => {
            el.addEventListener('mousedown', () => {
                if (window.navigator.vibrate) window.navigator.vibrate(10);
            });
        });

        // Range input sync
        const range = document.querySelector('input[type="range"]');
        const numInput = document.querySelector('input[placeholder="75"]');
        
        if (range && numInput) {
            range.addEventListener('input', (e) => numInput.value = e.target.value);
            numInput.addEventListener('input', (e) => range.value = e.target.value);
        }

        // Form submission animation
        document.querySelector('form').addEventListener('submit', (e) => {
            e.preventDefault();
            const btn = document.querySelector('.btn-primary');
            btn.innerHTML = '<i data-lucide="loader-2" class="w-6 h-6 animate-spin"></i> Saving...';
            lucide.createIcons();
            
            setTimeout(() => {
                btn.classList.replace('btn-primary', 'bg-accent');
                btn.style.background = 'var(--accent)';
                btn.innerHTML = '<i data-lucide="check" class="w-6 h-6"></i> Confirmed';
                lucide.createIcons();
            }, 1200);
        });
    </script>
