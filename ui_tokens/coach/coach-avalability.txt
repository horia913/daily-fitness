    <script>
        // Initialize Lucide
        lucide.createIcons();

        // Toggle Slot Availability
        function toggleSlot(el) {
            const isAvailable = el.classList.contains('available');
            
            if (window.navigator.vibrate) {
                window.navigator.vibrate(15);
            }

            if (isAvailable) {
                el.classList.remove('available');
                el.innerHTML = '';
            } else {
                el.classList.add('available');
                el.innerHTML = '<i data-lucide="check" class="w-4 h-4 text-secondary"></i>';
                lucide.createIcons();
            }
        }

        // Entrance Animation
        document.addEventListener('DOMContentLoaded', () => {
            const sections = document.querySelectorAll('section, header');
            sections.forEach((sec, i) => {
                sec.style.opacity = '0';
                sec.style.transform = 'translateY(15px)';
                setTimeout(() => {
                    sec.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    sec.style.opacity = '1';
                    sec.style.transform = 'translateY(0)';
                }, i * 120);
            });
        });
    </script>
