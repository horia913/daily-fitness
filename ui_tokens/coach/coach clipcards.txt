    <script>
        lucide.createIcons();

        function switchTab(tab) {
            const typesBtn = document.getElementById('tab-types');
            const assignBtn = document.getElementById('tab-assignments');
            const typesContent = document.getElementById('content-types');
            const assignContent = document.getElementById('content-assignments');
            const actionBtnText = document.getElementById('actionText');

            if (tab === 'types') {
                typesBtn.classList.add('tab-active', 'text-white');
                typesBtn.classList.remove('text-neutral-500');
                assignBtn.classList.remove('tab-active', 'text-white');
                assignBtn.classList.add('text-neutral-500');
                
                typesContent.classList.remove('hidden');
                assignContent.classList.add('hidden');
                actionBtnText.innerText = "New Package Type";
            } else {
                assignBtn.classList.add('tab-active', 'text-white');
                assignBtn.classList.remove('text-neutral-500');
                typesBtn.classList.remove('tab-active', 'text-white');
                typesBtn.classList.add('text-neutral-500');
                
                assignContent.classList.remove('hidden');
                typesContent.classList.add('hidden');
                actionBtnText.innerText = "Assign Clipcard";
            }

            // Haptic Feedback
            if (window.navigator.vibrate) {
                window.navigator.vibrate(5);
            }
        }

        // Subtle Entry Reveal
        document.addEventListener('DOMContentLoaded', () => {
            const cards = document.querySelectorAll('.crystal-card');
            cards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 50);
            });
        });
    </script>
