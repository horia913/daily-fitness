    <script>
        // Initialize Lucide
        lucide.createIcons();

        // Haptic Feedback Simulation
        document.querySelectorAll('button, .interactive-card, select').forEach(el => {
            el.addEventListener('click', () => {
                if (window.navigator.vibrate) {
                    window.navigator.vibrate(10);
                }
            });
        });

        // Dynamic interactivity for selects
        document.querySelectorAll('select').forEach(select => {
            select.addEventListener('change', function() {
                const card = this.closest('.interactive-card');
                const dot = card.querySelector('.w-2');
                if (this.value === 'Rest Day') {
                    dot.classList.remove('bg-primary');
                    dot.classList.add('bg-neutral-700');
                    card.style.opacity = '0.7';
                } else {
                    dot.classList.add('bg-primary');
                    dot.classList.remove('bg-neutral-700');
                    card.style.opacity = '1';
                }
            });
        });
