    <script>
        // Initialize Lucide
        lucide.createIcons();

        // Staggered entrance animation
        document.addEventListener('DOMContentLoaded', () => {
            const cards = document.querySelectorAll('.crystal-card');
            cards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 80);
            });
        });

        // Collapsible Logic
        function toggleCollapsible() {
            const section = document.querySelector('.collapsible-section');
            const icon = document.getElementById('collapse-icon');
            section.classList.toggle('collapsible-active');
            icon.style.transform = section.classList.contains('collapsible-active') ? 'rotate(180deg)' : 'rotate(0deg)';
        }

        // Haptic Feedback for FAB
        function openCreateModal() {
            if (window.navigator.vibrate) {
                window.navigator.vibrate([20, 10, 20]);
            }
            // Mock Modal Call
            alert("Open Full-Screen Create Challenge Modal");
        }

        // Mock Swipe Simulation (Visual only)
        document.querySelectorAll('.crystal-card').forEach(card => {
            let startX;
            card.addEventListener('touchstart', e => startX = e.touches[0].clientX);
            card.addEventListener('touchend', e => {
                let diff = startX - e.changedTouches[0].clientX;
                if (diff > 50) {
                    card.style.transform = 'translateX(-80px)';
                    setTimeout(() => card.style.transform = 'translateX(0)', 1500);
                }
            });
        });
    </script>
