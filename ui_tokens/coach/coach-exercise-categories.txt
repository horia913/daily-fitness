        // Initialize Lucide icons
        lucide.createIcons();

        // Reveal animation for cards
        document.addEventListener('DOMContentLoaded', () => {
            const cards = document.querySelectorAll('.crystal-card, .border-dashed');
            cards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(30px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 80);
            });
        });

        // Haptic feedback for primary actions
        const buttons = document.querySelectorAll('.btn-primary, .fab-mobile');
        buttons.forEach(btn => {
            btn.addEventListener('click', () => {
                if (window.navigator.vibrate) {
                    window.navigator.vibrate(10);
                }
                console.log('Action: Create New Category Modal Triggered');
            });
        });

        // Simulating edit/delete feedback
        document.querySelectorAll('.btn-ghost').forEach(btn => {
            btn.addEventListener('click', (e) => {
                e.stopPropagation();
                if (window.navigator.vibrate) window.navigator.vibrate(5);
            });
        });
