    <script>
        // Initialize Icons
        lucide.createIcons();

        // Basic interactivity for Protocol Chips
        document.querySelectorAll('.protocol-chip').forEach(chip => {
            chip.addEventListener('click', () => {
                document.querySelectorAll('.protocol-chip').forEach(c => c.classList.remove('active'));
                chip.classList.add('active');
                
                // Haptic feedback
                if (window.navigator.vibrate) window.navigator.vibrate(10);
            });
        });

        // Animation for entrance
        document.addEventListener('DOMContentLoaded', () => {
            const cards = document.querySelectorAll('.crystal-card');
            cards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 80);
            });
        });

        // Highlight input changes
        document.querySelectorAll('input, textarea').forEach(el => {
            el.addEventListener('change', () => {
                const card = el.closest('.crystal-card');
                if (card) {
                    card.classList.add('is-dirty');
                    if (!card.querySelector('.change-badge')) {
                        const title = card.querySelector('h3');
                        const badge = document.createElement('span');
                        badge.className = 'change-badge';
                        badge.innerText = 'Edited';
                        title.appendChild(badge);
                    }
                }
            });
        });
    </script>
