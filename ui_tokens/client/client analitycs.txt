    <script>
        lucide.createIcons();

        // Animate the progress rings on load
        document.addEventListener('DOMContentLoaded', () => {
            setTimeout(() => {
                document.querySelectorAll('.stat-ring').forEach(ring => {
                    const offset = ring.getAttribute('style').match(/dashoffset: (\d+)/)[1];
                    ring.style.strokeDashoffset = offset;
                });
            }, 300);

            // Subtle parallax on cards
            document.addEventListener('mousemove', (e) => {
                const cards = document.querySelectorAll('.vitreous-card');
                const x = (e.clientX / window.innerWidth - 0.5) * 15;
                const y = (e.clientY / window.innerHeight - 0.5) * 15;
                
                cards.forEach(card => {
                    card.style.transform = `perspective(1000px) rotateX(${-y}deg) rotateY(${x}deg)`;
                });
            });
        });
    </script>
