    <script>
        // Initialize Lucide icons
        lucide.createIcons();

        // Accordion Toggle
        function toggleAccordion() {
            const acc = document.getElementById('rulesAccordion');
            acc.classList.toggle('accordion-active');
        }

        // Mock Countdown Timer
        function updateCountdown() {
            const display = document.getElementById('countdown');
            // Simplified logic for demo
            let now = new Date().getTime();
            let target = now + (3 * 24 * 60 * 60 * 1000) + (14 * 60 * 60 * 1000); // 3d 14h from now
            
            setInterval(() => {
                let current = new Date().getTime();
                let diff = target - current;
                
                let d = Math.floor(diff / (1000 * 60 * 60 * 24));
                let h = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                let m = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
                let s = Math.floor((diff % (1000 * 60)) / 1000);
                
                display.innerText = `${d}d ${h}h ${m}m ${s}s`;
            }, 1000);
        }

        // Entry Animations
        document.addEventListener('DOMContentLoaded', () => {
            updateCountdown();
            
            const cards = document.querySelectorAll('.crystal-card, section h3, .podium-gradient-1, .podium-gradient-2, .podium-gradient-3');
            cards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(30px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.8s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 80);
            });
        });

        // Haptic Feedback Simulation
        document.querySelectorAll('button').forEach(btn => {
            btn.addEventListener('click', () => {
                if (window.navigator.vibrate) {
                    window.navigator.vibrate(15);
                }
            });
        });
    </script>
