    <script>
        lucide.createIcons();

        function toggleSheet() {
            document.getElementById('time-sheet').classList.toggle('open');
        }

        function selectDate(element) {
            // Check if day is Jan 5 (which is already booked)
            if(element.innerText === '5') {
                document.getElementById('limit-alert').classList.remove('hidden');
                setTimeout(() => {
                    document.getElementById('limit-alert').classList.add('hidden');
                }, 4000);
                return;
            }

            // UI feedback
            document.querySelectorAll('.day-cell').forEach(d => d.classList.remove('active'));
            element.classList.add('active');
            
            // Open sheet
            document.getElementById('time-sheet').classList.add('open');
        }

        function selectTime(element) {
            document.querySelectorAll('.time-chip').forEach(t => {
                t.classList.remove('selected');
                t.querySelector('.text-neutral-100')?.classList.replace('text-neutral-100', 'text-neutral-400');
            });
            element.classList.add('selected');
        }

        function confirmBooking() {
            const btn = document.querySelector('.btn-primary');
            const originalText = btn.innerHTML;
            btn.innerHTML = `<i data-lucide="loader-2" class="w-6 h-6 animate-spin"></i> Processing...`;
            lucide.createIcons();

            if (window.navigator.vibrate) {
                window.navigator.vibrate([20, 50, 20]);
            }

            setTimeout(() => {
                btn.innerHTML = `<i data-lucide="check" class="w-6 h-6"></i> Confirmed!`;
                btn.style.background = 'linear-gradient(135deg, #10B981 0%, #059669 100%)';
                btn.style.boxShadow = '0 8px 24px rgba(16, 185, 129, 0.3)';
                lucide.createIcons();
                
                setTimeout(() => {
                    toggleSheet();
                    // Reset button for demo
                    setTimeout(() => {
                        btn.innerHTML = originalText;
                        btn.style.background = '';
                        btn.style.boxShadow = '';
                        lucide.createIcons();
                    }, 1000);
                }, 1500);
            }, 2000);
        }

        // Animation entry
        document.addEventListener('DOMContentLoaded', () => {
            const sections = document.querySelectorAll('section, header');
            sections.forEach((s, i) => {
                s.style.opacity = '0';
                s.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    s.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    s.style.opacity = '1';
                    s.style.transform = 'translateY(0)';
                }, i * 150);
            });
        });
    </script>
