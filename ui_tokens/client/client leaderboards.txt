    <script>
        // Initialize Lucide Icons
        lucide.createIcons();

        // Ghost Mode Toggle Logic
        const anonToggle = document.getElementById('anon-toggle');
        anonToggle.addEventListener('click', () => {
            anonToggle.classList.toggle('toggle-active');
            if (window.navigator.vibrate) {
                window.navigator.vibrate(10);
            }
        });

        // Filter Chip Interaction
        const chips = document.querySelectorAll('.filter-chip');
        chips.forEach(chip => {
            chip.addEventListener('click', () => {
                chips.forEach(c => {
                    c.classList.remove('active');
                    c.classList.add('text-neutral-400');
                });
                chip.classList.add('active');
                chip.classList.remove('text-neutral-400');
                if (window.navigator.vibrate) {
                    window.navigator.vibrate(5);
                }
            });
        });

        // Entrance Animation
        document.addEventListener('DOMContentLoaded', () => {
            const cards = document.querySelectorAll('.crystal-card, .podium-gradient-1, .podium-gradient-2, .podium-gradient-3');
            cards.forEach((card, index) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, index * 80);
            });
        });
    </script>
