    <script>
        // Initialize Lucide icons
        lucide.createIcons();

        // Simple Tab Switcher
        function switchTab(el, tabName) {
            // Update UI
            const buttons = document.querySelectorAll('.tab-btn');
            buttons.forEach(btn => {
                btn.classList.remove('active', 'text-white');
                btn.classList.add('text-neutral-500');
            });
            el.classList.add('active', 'text-white');
            el.classList.remove('text-neutral-500');

            // Simulate content change
            const list = document.getElementById('session-list');
            list.style.opacity = '0';
            list.style.transform = 'translateY(10px)';
            
            setTimeout(() => {
                list.style.transition = 'all 0.5s ease';
                list.style.opacity = '1';
                list.style.transform = 'translateY(0)';
                // In a real app, you would filter data here
            }, 200);
        }

        // Haptic Feedback for primary actions
        document.querySelectorAll('.btn-primary, .fab-pulse').forEach(btn => {
            btn.addEventListener('click', () => {
                if (window.navigator.vibrate) {
                    window.navigator.vibrate(15);
                }
            });
        });

        // Swipe handling placeholder (Mobile optimization)
        let touchstartX = 0;
        let touchendX = 0;
        
        function checkDirection() {
          if (touchendX < touchstartX - 100) {
            // Swiped left - go to history
            console.log('Swiped Left');
          }
          if (touchendX > touchstartX + 100) {
            // Swiped right - go to upcoming
            console.log('Swiped Right');
          }
        }

        document.addEventListener('touchstart', e => {
          touchstartX = e.changedTouches[0].screenX;
        });

        document.addEventListener('touchend', e => {
          touchendX = e.changedTouches[0].screenX;
          checkDirection();
        });
    </script>
