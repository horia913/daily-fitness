    <script>
        // Initialize Lucide icons
        lucide.createIcons();

        // Entry Animations
        document.addEventListener('DOMContentLoaded', () => {
            const modal = document.querySelector('.crystal-modal');
            const sections = document.querySelectorAll('section');
            
            modal.style.opacity = '0';
            modal.style.transform = 'translateY(100px)';
            
            setTimeout(() => {
                modal.style.transition = 'all 0.8s cubic-bezier(0.16, 1, 0.3, 1)';
                modal.style.opacity = '1';
                modal.style.transform = 'translateY(0)';
            }, 100);

            sections.forEach((s, i) => {
                s.style.opacity = '0';
                s.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    s.style.transition = 'all 0.6s cubic-bezier(0.16, 1, 0.3, 1)';
                    s.style.opacity = '1';
                    s.style.transform = 'translateY(0)';
                }, 300 + (i * 100));
            });
        });

        // Haptic Feedback Simulation
        const saveBtn = document.getElementById('save-btn');
        saveBtn.addEventListener('click', () => {
            if (window.navigator.vibrate) {
                window.navigator.vibrate(20);
            }
            
            saveBtn.innerHTML = `<i data-lucide="loader-2" class="w-5 h-5 animate-spin"></i> SAVING...`;
            lucide.createIcons();
            
            // Mock Success State
            setTimeout(() => {
                saveBtn.classList.remove('btn-success');
                saveBtn.style.background = '#10B981';
                saveBtn.innerHTML = `<i data-lucide="check" class="w-5 h-5"></i> SAVED!`;
                lucide.createIcons();
                
                setTimeout(() => {
                    // Logic to return to previous page
                    console.log("Food saved successfully");
                }, 1000);
            }, 1500);
        });

        // Interactive Macro Visualization (Mockup)
        const inputs = document.querySelectorAll('input[type="number"]');
        inputs.forEach(input => {
            input.addEventListener('focus', () => {
                if (window.navigator.vibrate) window.navigator.vibrate(5);
            });
        });
    </script>
