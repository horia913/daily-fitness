    <script>
        // Initialize Lucide
        lucide.createIcons();

        // Toggle collapsible exercise groups
        function toggleCollapsible(btn) {
            const group = btn.parentElement;
            const icon = btn.querySelector('[data-lucide="chevron-down"]');
            
            // Close others (optional)
            /*
            document.querySelectorAll('.collapsible-group').forEach(other => {
                if(other !== group) other.classList.remove('collapsible-active');
            });
            */

            group.classList.toggle('collapsible-active');
            
            if(group.classList.contains('collapsible-active')) {
                icon.style.transform = 'rotate(180deg)';
            } else {
                icon.style.transform = 'rotate(0deg)';
            }
        }

        // Initialize first one as active for visual demo
        document.querySelector('.collapsible-group').classList.add('collapsible-active');

        // Filter pills interaction
        document.querySelectorAll('.filter-pill').forEach(pill => {
            pill.addEventListener('click', () => {
                document.querySelectorAll('.filter-pill').forEach(p => p.classList.remove('active'));
                pill.classList.add('active');
            });
        });

        // Vibrate feedback for Share Button
        document.querySelector('.btn-share').addEventListener('click', () => {
            if (window.navigator.vibrate) {
                window.navigator.vibrate([20, 10, 20]);
            }
        });
    </script>
