    <script>
        // Initialize Icons
        lucide.createIcons();

        function switchTab(section, el) {
            // Update UI
            document.querySelectorAll('.tab-trigger').forEach(btn => btn.classList.remove('active'));
            el.classList.add('active');

            // Move Indicator
            const indicator = document.getElementById('tab-indicator');
            indicator.style.width = `${el.offsetWidth}px`;
            indicator.style.transform = `translateX(${el.offsetLeft - 6}px)`;

            // Switch Visibility
            if (section === 'goals') {
                document.getElementById('section-goals').classList.remove('hide');
                document.getElementById('section-habits').classList.add('hide');
            } else {
                document.getElementById('section-goals').classList.add('hide');
                document.getElementById('section-habits').classList.remove('hide');
            }

            // Haptic feedback
            if (window.navigator.vibrate) window.navigator.vibrate(10);
        }

        // Initialize indicator position
        window.addEventListener('load', () => {
            const activeTab = document.querySelector('.tab-trigger.active');
            const indicator = document.getElementById('tab-indicator');
            indicator.style.width = `${activeTab.offsetWidth}px`;
            indicator.style.transform = `translateX(${activeTab.offsetLeft - 6}px)`;

            // Subtle entry animation for cards
            const cards = document.querySelectorAll('.crystal-card');
            cards.forEach((card, i) => {
                card.style.opacity = '0';
                card.style.transform = 'translateY(20px)';
                setTimeout(() => {
                    card.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    card.style.opacity = '1';
                    card.style.transform = 'translateY(0)';
                }, i * 100);
            });
        });
    </script>
