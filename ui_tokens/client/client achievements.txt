    <script>
        // Initialize Lucide
        lucide.createIcons();

        function openModal(title, tier, desc) {
            const modal = document.getElementById('badge-modal');
            const iconCont = document.getElementById('modal-icon-container');
            
            document.getElementById('modal-title').innerText = title;
            document.getElementById('modal-tier').innerText = tier;
            document.getElementById('modal-desc').innerText = desc;

            // Simple Tier coloring for modal icon background
            if(tier.includes('Gold')) {
                iconCont.className = "w-24 h-24 mx-auto mb-6 rounded-3xl flex items-center justify-center shadow-2xl tier-gold achievement-float";
            } else if(tier.includes('Silver')) {
                iconCont.className = "w-24 h-24 mx-auto mb-6 rounded-3xl flex items-center justify-center shadow-2xl tier-silver achievement-float";
            } else {
                iconCont.className = "w-24 h-24 mx-auto mb-6 rounded-3xl flex items-center justify-center shadow-2xl tier-bronze achievement-float";
            }

            modal.classList.remove('hidden');
            if (window.navigator.vibrate) window.navigator.vibrate(20);
        }

        function closeModal() {
            document.getElementById('badge-modal').classList.add('hidden');
        }

        // Animated Entry Logic
        document.addEventListener('DOMContentLoaded', () => {
            const gridItems = document.querySelectorAll('.badge-grid-item');
            gridItems.forEach((item, index) => {
                item.style.opacity = '0';
                item.style.transform = 'translateY(20px) scale(0.9)';
                setTimeout(() => {
                    item.style.transition = 'all 0.6s cubic-bezier(0.2, 0.8, 0.2, 1)';
                    item.style.opacity = '1';
                    item.style.transform = 'translateY(0) scale(1)';
                }, index * 80);
            });
        });
    </script>
